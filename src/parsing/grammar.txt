<program> ::= { <declaration> } EOF ;
<declaration> ::= <varDecl> | <letBinding> | <statement>
<varDecl> ::= "var" ID ( = <expr> )? ;
<letBinding> ::= "let" ID = <expr> ;
<statement> ::= <exprStmt> | <printStmt>
<exprStmt> ::= <expr> ;
<printStmt> ::= <expr>

expressions
...
<unary> ::= "-" <unary> | <application>
<application> ::= <primary> { ( <arguments>? ) }
<arguments> ::= <expression> { , <expression> }
